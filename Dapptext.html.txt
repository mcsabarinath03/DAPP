  <script>
    // â›“ï¸ Paste your new deployed contract address here
    const contractAddress = "PASTE_YOUR_NEW_CONTRACT_ADDRESS";

    // ðŸ§  Updated ABI for string storage
    const contractABI = [
      {
        "inputs": [{"internalType": "string", "name": "_msg", "type": "string"}],
        "name": "setMessage",
        "outputs": [],
        "stateMutability": "nonpayable",
        "type": "function"
      },
      {
        "inputs": [],
        "name": "getMessage",
        "outputs": [{"internalType": "string", "name": "", "type": "string"}],
        "stateMutability": "view",
        "type": "function"
      }
    ];

    let provider, signer, contract;

    async function connectWallet() {
      if (window.ethereum) {
        provider = new ethers.BrowserProvider(window.ethereum);
        await provider.send("eth_requestAccounts", []);
        signer = await provider.getSigner();
        contract = new ethers.Contract(contractAddress, contractABI, signer);
        console.log("Connected to MetaMask");
      } else {
        alert("Please install MetaMask!");
      }
    }

    async function setMessage() {
      const msg = document.getElementById("newMessage").value;
      if (!msg) return alert("Please enter a message!");
      const tx = await contract.setMessage(msg);
      await tx.wait();
      alert("Message stored successfully!");
    }

    async function getMessage() {
      const result = await contract.getMessage();
      document.getElementById("currentMessage").innerText = result;
    }

    document.getElementById("setButton").addEventListener("click", async () => {
      await connectWallet();
      await setMessage();
    });

    document.getElementById("getButton").addEventListener("click", async () => {
      await connectWallet();
      await getMessage();
    });
  </script>
